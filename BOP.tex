\begin{frame}
\frametitle{Увод}
\framesubtitle{}
\begin{itemize}
\item Интересува ни да измерим икономическите взаимоотношения между нашата икономика и останалите държави (\emph{останал свят})
\item Основни статистически източници:
\begin{itemize}
\item Платежен баланс
\item Международна инвестиционна позиция
\item Статистика на външния дълг
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Потоци и размери (запаси)}
\framesubtitle{}
\begin{itemize}
\item Размери (запаси)
\begin{itemize}
\item Регистрират се към даден момент от времето (на дадена дата)
\item Дават ,,моментна снимка''
\item Примери: стока в склада на една фирма в края на даден ден, сумата от
спестяванията на даден човек в края на годината, наличното количество пари в икономиката в определен момент
\end{itemize}
\item Потоци
\begin{itemize}
\item Задават се за период от време
\item Показват изменение за периода
\item Примери: произведената продукция за определено тримесечие, месечната заплата, отпуснатите от една банка заеми за дадена седмица
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Потоци и размери (запаси)}
\framesubtitle{}
\begin{itemize}
\item Между потоци и размери има връзка: изменението на даден размер между два момента от времето задава поток (който на свой ред може да се състои от други потоци)
\item За външния сектор:
\begin{itemize}
\item Платежен баланс -- потоци
\item Международна инвестиционна позиция и външен дълг -- размери
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Платежен баланс}
\framesubtitle{Общи положения}
\begin{itemize}
\item В платежния баланс се записват стойностите на транзакциите между местни и чуждестранни икономически агенти за определен период от време
\item Има транзакции с реален и с финансов характер
\item Транзакциите са свързани с входящи или изходящи парични потоци
\item Най-общо платежният баланс се подразделя на две части
\begin{itemize}
\item текуща сметка
\item капиталова и финансова сметка
\end{itemize}
\end{itemize}
\end{frame}


\begin{frame}
\frametitle{Основни връзки в платежния баланс}
\framesubtitle{}
Основно тъждество:
\beq \textrm{CA} + \textrm{FKA} \equiv 0,  \label{eq:BoPidentity1}\eeq
където $\textrm{CA}$ -- нетен баланс на транзакциите по текущата сметка, $\textrm{FKA}$ -- нетен баланс на транзакциите по капиталовата и финансовата сметка

\medskip Текуща сметка:
\beq \textrm{CA}:=X-M+Y_f+\textrm{Tr}_f \label{eq:CA1}\eeq

Текущата сметка може още да бъде записана като \beq
\textrm{CA}=\textrm{GNDI}-C-I-G = \textrm{GNDI}-A ,
\label{eq:CA2}\eeq където $A:=C+I+G$ се нарича
\emph{абсорбция}
\end{frame}

\begin{frame}
\frametitle{Основни връзки в платежния баланс}
\framesubtitle{}
Вътрешни спестявания: \beq S^d := Y-C-G  \label{eq:S_dom}\eeq

Национални спестявания:  \beq S := \textrm{GNDI}-C-G = Y+Y_f+\textrm{Tr}_f-C-G  \label{eq:S_natl}\eeq

Тогава за текущата сметка имаме: \beq \textrm{CA}=S-I \label{eq:CA3}\eeq
\end{frame}

\begin{frame}
\frametitle{Стандартно и аналитично представяне на платежния баланс}
\framesubtitle{}
\begin{itemize}
\item Основното тъждество на платежния баланс е свързано с т.нар. \emph{стандартно представяне}
\item Стандартното представяне подчертава счетоводните принципи и връзките между реални и финансови транзакции, но може да не е толкова удобно за анализ на икономическите процеси
\item За целта от него се конструират представяния на платежния баланс, които са подходящи за аналитични цели, т.нар. \emph{аналитични представяния}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Стандартно и аналитично представяне на платежния баланс}
\framesubtitle{}
\begin{itemize}
\item При аналитичните представяния най-често текущата сметка се запазва, а разделение се въвежда за капиталовата и финансовата сметка
\item Популярен вариант е да се създаде нов раздел, включващ изменението на валутните резерви и част от целевото финансиране за макроикономическа стабилизация, получено от международни институции
\end{itemize}
\beq \textrm{CA} + \textrm{FKA} - \Delta \textrm{Res} \equiv 0 
\label{eq:BoPidentity2}\eeq

\beq \textrm{FKA} = \textrm{KA} + \textrm{FDI} + \textrm{PortfInv} +
\textrm{OtherInv}. \label{eq:FA}\eeq
\end{frame}

\begin{frame}
\frametitle{Външен дълг}
\framesubtitle{}
\begin{itemize}
\item Външният дълг включва стойността на всички задължения на местни икономически агенти към нерезиденти към даден момент
\item Задълженията могат да включват бъдещи плащания както на главници, така и на лихви
\item Примери 
\begin{itemize}
\item емитирани от местни лица облигации, закупени от нерезиденти
\item директни заеми от чуждестранни икономически агенти
\item задължения по търговски кредити
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Външен дълг}
\framesubtitle{}
\begin{itemize}
\item \emph{Брутен външен дълг} -- отчита колко дадена икономика дължи на останалия свят, но не и какви са задълженията на нерезиденти към съответната икономика
\item \emph{Нетен външен дълг} -- размерът на брутния външен дълг, намален с размера на задълженията на чужди икономически агенти към националната икономика
\item Има различни представяния (разбивки) на брутния външен дълг
\begin{itemize}
\item според институционалните сектори, които са длъжници
\item по кредитори (международни организации, чужди правителства, чуждестранни банки, други нерезиденти)
\item по инструменти (ценни книжа, заеми, търговски кредити, депозити на нерезиденти и пр.)
\item по матуритет (краткосрочен -- с падеж до 1 година и дългосрочен -- с падеж над 1 година)
\item по географски признак (по страни/региони на кредиторите)
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Международна инвестиционна позиция}
\framesubtitle{}
\begin{itemize}
\item Дава стойността на активите и пасивите на една икономика по отношение на останалия свят към даден момент
\begin{itemize}
\item външни активи на една икономика -- права на икономическите агенти да получат в бъдещето плащания от нерезиденти
\item външни пасиви -- права на нерезиденти да получат плащания от местни икономически агенти
\end{itemize}
\item Структурата на активната и пасивната страна на международната инвестиционна позиция е подобна на структурата на финансовата сметка на платежния баланс...
\item МИП към две дати и финансовата сметка на платежния баланс за съответния период дават размери и съответстващия им поток
\end{itemize}
\end{frame}
