\begin{frame}
\frametitle{Нетен вътрешен продукт}
\framesubtitle{}
Ако от общото производство махнем частта, която е била използвана, за
да замени изхабен капитал, остава ,,чисто'' производство $\rightarrow$ \textit{нетен вътрешен продукт}

\beq \begin{split} \textrm{нетен вътрешен продукт} = &
\textrm{ брутен вътрешен продукт} - \\
 & \textrm{ амортизация}\end{split} \label{eq:NNP}\eeq
\end{frame}


\begin{frame}
\frametitle{Разходен метод}
\framesubtitle{}
Разглежда БВП от гледна точка на това за какво е използвано произведеното. Понеже използването най-често е свързано с правене на разходи -- \textit{разходен метод}.\pause

\textbf{Основна логика на разделението в рамките на разходния метод:}
\begin{enumerate}
\item Произведеното може да бъде използвано в националната икономика или да бъде изнесено \pause
\item Използването в националната икономика може да бъде за текуща употреба (потребление) или да влияе върху бъдещия производствен потенциал (инвестиции) \pause
\item Използването (или за потребление, или за инвестиции) може да бъде от частния сектор или от държавата \pause
\item Трябва да се отчита, че част от разходите за потребление или инвестиции отива за вносна продукция (т.е. не е наше производство)
\end{enumerate}
\end{frame}

\begin{frame}
\frametitle{Номинален и реален БВП}
\framesubtitle{}
\begin{itemize}
\item Разглежданите дотук показатели представляват стойности т.е. количества,
умножени по цените за съответния период $\rightarrow$ показатели в номинално изражение
\item Такъв показател е и БВП в различните му представяния: номинален БВП, сумира стойности
\item Тогава неговите изменения за различни периоди във времето се дължат както на изменения в количествата, така и на ценови изменения
\item Това може да създаде проблеми в интерпретацията: напр. номиналният БВП се е увеличил, но това е изцяло заради цените
\item  Често ни интересува колко са произведените количества $\Rightarrow$ необходимо е номиналният БВП да бъде коригиран, за да се елиминира изменението на цените (привеждане по
постоянни цени, дефлиране)
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Основни формули}
\framesubtitle{Номинален растеж, реален растеж, дефлатор}

\beq 1+n_t=(1+g_t)(1+\pi_t), \label{eq:GDPdeflGrth}\eeq където
с $n_t$ е означено номиналното процентно изменение на БВП между периодите $t$ и $t-1$, $g_t$ е реалният растеж между двата периода, а с $\pi_t$ е означен дефлаторът.
\end{frame}

\begin{frame}
\frametitle{Основни формули}
\framesubtitle{Изчисление на база компоненти}
\begin{itemize}
\item В икономиката се произвеждат $m$ стоки
\item Произведеното количество от $i$-тата стока в период $t$ означаваме с $q_{i,t}$, а съответната цена е $p_{i,t}$
\item Номиналната стойност на произведеното в период $t-1$ (т.е. по текущи цени) е $\sum_{i=1}^m p_{i,t-1} q_{i,t-1}$
\item Номиналната стойност на произведеното в период $t$ (отново по текущи цени) е $\sum_{j=1}^m p_{j,t} q_{j,t}$
\item Стойността на произведеното в период $t$ по цени от период $t-1$ е $\sum_{j=1}^m p_{j,t-1} q_{j,t}$
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Основни формули}
\framesubtitle{Изчисление на база компоненти}
\textbf{Реално изменение $g_t$ на производството между двата периода:} \[ \underbrace{\frac{\sum_{j=1}^m p_{j,t-1} q_{j,t}}{\sum_{i=1}^m p_{i,t-1} q_{i,t-1}}}_{1+g_t} = \sum_{j=1}^m \underbrace{\frac{p_{j,t-1} q_{j,t-1}}{\sum_{i=1}^m p_{i,t-1} q_{i,t-1}}}_{w_j} \cdot \underbrace{\frac{q_{j,t}}{q_{j,t-1}}}_{1+g_{j,t}}= \sum_{j=1}^m w_j (1+g_{j,t})  \]

Т.е. реалното изменение на БВП може да се получи чрез претегляне на растежите $1+g_{j,t}$ на компонентите му в индексна форма при конструиране
на подходящите тегла $w_j$ (номиналната структура на производството от предходния период).
\end{frame}

\begin{frame}
\frametitle{Основни формули}
\framesubtitle{Изчисление на база компоненти}
\textbf{Номинално изменение $n_t$ на производството:}
\[ \underbrace{\frac{\sum_{j=1}^m p_{j,t} q_{j,t}}{\sum_{i=1}^m p_{i,t-1} q_{i,t-1}}}_{1+n_t} = \underbrace{\frac{\sum_{k=1}^m p_{k,t-1} q_{k,t}}{\sum_{i=1}^m p_{i,t-1} q_{i,t-1}}}_{1+g_t} \cdot \underbrace{\frac{\sum_{j=1}^m p_{j,t} q_{j,t}}{\sum_{k=1}^m p_{k,t-1} q_{k,t}}}_{1+\pi_t}  \]

Тук, както по-горе, дефинираме като остатък дефлатор $1+\pi_t$.
\end{frame}

\begin{frame}
\frametitle{Основни формули}
\framesubtitle{Изчисление на база компоненти}
\textbf{Дефлатор $\pi_t$ :}  \[ \underbrace{\frac{\sum_{i=1}^m p_{i,t} q_{i,t}}{\sum_{j=1}^m p_{j,t-1} q_{j,t}}}_{1+\pi_t}= \sum_{i=1}^m \underbrace{\frac{p_{i,t-1} q_{i,t}}{\sum_{j=1}^m p_{j,t-1} q_{j,t}}}_{\tilde{w}_i} \cdot \underbrace{\frac{p_{i,t}}{p_{i,t-1}}}_{1+\pi_{i,t}} = \sum_{i=1}^m \tilde{w}_i (1+\pi_{i,t}) \]

\begin{itemize}
\item Общият дефлатор на БВП се получава чрез претегляне на дефлаторите
на компонентите $\pi_{i,t}$ в индексна форма, където теглата $\tilde{w}_i$
се получават от структурата на производството за текущия период, но оценена
по цени от предходния период.
\item  Произволен ценови индекс има структурата, показана на края на горната формула, като разлики има в начина на конструиране на теглата.
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Основни формули}
\begin{itemize}
\item В практиката първо се изчисляват дефлатори и с тях се дефлират номиналните величини, за да се получат величини по постоянни цени
\item Принципите за изчисление на база компоненти са приложими и към различните методи на изчисляване на БВП 
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Национален доход}
\begin{itemize}
\item Доходите, получавани от икономическите агенти, са важна макроикономическа величина
\item Производството в икономиката е основен, но не и единствен
източник на доходи
\item Други източници на доходи са:
\begin{enumerate}
  \item \textit{Доходи от работа в чужбина.} Те не са част
      от БВП, тъй като производството е било в рамките на
      друга икономика.
  \item \textit{Доходи от инвестиции в чуждестранни ценни
      книги и инструменти.} $\rightarrow$ лихви и дивиденти,
      получавани от местни икономически агенти от
      притежавани от тях чуждестранни ценни книжа.
      
     Такива ценни книги най-често са придобити със
      спестени пари през минали периоди $\rightarrow$
      могат да се разглеждат като възвращаемост от
      спестяванията, направени в икономиката в миналото.
  \item \textit{Безвъзмездни постъпления на пари от други
      икономики.}  $\rightarrow$ дарения, направени от
      чуждестранни частни лица или като официална
      правителствена помощ за дадена икономика (напр.
      еврофондове).
\end{enumerate}
\end{itemize}
\end{frame}


\begin{frame}
\frametitle{Национален доход}
\begin{itemize}
\item И при трите вида източници ресурсите се отчитат на нетна база
т.е. $\textrm{постъпления} - \textrm{плащания}$
\item Пример: доходите, получени от български граждани от работа в чужбина,
трябва да се намалят със сумата на изплатените на чужденци възнаграждения за работа в България
\item Първите два източника на доходи се наричат \textit{първичен доход}. Старо наименование: \textit{нетен факторен доход от чужбина}, понеже те могат да се интерпретират като възнаграждение
за предоставяне на производствени фактори на чуждестранни
икономически агенти (труд и индиректно капитал
чрез спестени финансови ресурси)
\item Първичният доход от чужбина може да
бъде и отрицателен: разлика между постъпления и
плащания на възнаграждения за съответните фактори
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Национален доход}
\begin{itemize}
\item Ако към стойността $Y$ на БВП прибавим първичния доход (означаван с $\text{BPI}$), ще получим т.нар. \textbf{брутен национален доход (БНД)}, означаван с GNI
\item За БНД се използва още и терминът \textbf{брутен национален продукт (БНП)}
\item  Имаме \beq \textrm{GNI} = Y + \text{BPI} .
\label{eq:GNI}\eeq
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Национален доход}
\begin{itemize}
\item Третият източник на доходи се нарича \textit{вторичен доход} (старо наименование: \textit{нетни текущи
трансфери})
\item  Вторичният доход също може да бъде отрицателен
\item Ако вторичният доход $\textrm{BSI}$ се прибави към БНД, получаваме т.нар.
\textbf{брутен национален разполагаем доход
(\mbox{БНРД})}, означаван с GNDI
\item Имаме \beq \textrm{GNDI} =
\textrm{GNI} + \textrm{BSI}  \label{eq:GNDI}\eeq
\end{itemize}
\end{frame}
