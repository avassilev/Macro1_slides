\begin{frame}
\frametitle{Краткосрочни и дългосрочни модели}
\framesubtitle{}
\begin{itemize}
\item Наблюдаван БВП = дългосрочен компонент + краткосрочен компонент
\item Дългосрочен компонент -- равновесната траектория на производството в
достатъчно голям хоризонт от време (\emph{потенциално производство, потенциален продукт, потенциален БВП})
\item Краткосрочен компонент  -- отклонение от потенциалното производство
\item Идеята за дългосрочен равновесен компонент и краткосрочни отклонения от него важи и за други променливи освен БВП
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Краткосрочни и дългосрочни модели}
\framesubtitle{Допълнения}
\begin{itemize}
\item Различни дефиниции на краткосрочен и дългосрочен хоризонт 
\begin{itemize}
\item краткосрочен -- толкова къс, че капиталът в икономиката не се
изменя; дългосрочен -- има изменения в размера на капитала
\item краткосрочен -- до 1 година, средносрочен --  от 1 до 3 години, дългосрочен -- над 3
години
\item и двата варианта имат недостатъци
\end{itemize}
\item Един макроикономически модел би трябвало да включва както явлението икономически растеж, така и бизнес цикъла. Разделението на моделите е условно и се прави за удобство.
\end{itemize}

\item От статистическа гледна точка потенциалният БВП е ненаблюдаем и само може да бъде оценен чрез статистически или теоретичен модел
\end{itemize}

\begin{frame}
\frametitle{Основен IS-MP-PC модел}
\framesubtitle{}
\begin{itemize}
\item Линеен модел
\item Три променливи
\begin{itemize}
\item отклонение на реалното производство от потенциалната му стойност
\item инфлация
\item реален лихвен процент
\end{itemize} 
\item $Y_t$ -- реализиран БВП за период $t$, $\bar{Y}_t$ -- потенциално производство за период $t$;
$$\hat{y}_t := \frac{Y_t-\bar{Y}_t}{\bar{Y}_t}$$ ce нарича
\emph{отклонение от потенциалното производство}
\end{itemize}
\end{frame}


\begin{frame}
\frametitle{Основен IS-MP-PC модел}
\framesubtitle{}
\begin{itemize}
\item  За избран ценови индекс $P_t$ имаме
$$\pi_t := \frac{P_t-P_{t-1}}{P_{t-1}},$$ което представлява \emph{инфлация} спрямо предходния период
\item $i_t$ -- номинален лихвен процент за период $t$;
$$r_t := \frac{1+i_t}{1+\pi_t}-1,$$ се нарича \emph{реален лихвен
процент} т.е. това е лихвен процент, коригиран с инфлацията
\item За малки стойности на инфлацията и номиналния лихвен процент може да се използва приближеното равенство
$$r_t \approx i_t - \pi_t$$
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Основен IS-MP-PC модел}
\framesubtitle{IS връзка}
\textbf{IS права}: \beq \hat{y}_t =A -a r_t+\varepsilon_{1,t}, \label{eq:ISline0}\eeq където
$A,a>0$ са коефициенти, а $\varepsilon_{1,t}$ е шок.

При $r_t=\bar{r}=A/a$ и при отсъствие на шок
($\varepsilon_{1,t}=0$), имаме $$Y_t=\bar{Y}_t \textrm{ и } \hat{y}_t=0.$$ Тогава производството е на равновесната си стойност и съответният
лихвен процент $\bar{r}$ ще наричаме \emph{{равновесен реален
лихвен процент}}
\end{frame}

\begin{frame}
\frametitle{Основен IS-MP-PC модел}
\framesubtitle{IS връзка}
\begin{itemize}
\item Нека $\varepsilon_{1,t}=0$. За $r_t>\bar{r}$ имаме $\hat{y}_t < 0$,
т.е. производството в икономиката е под потенциалното 
\item  При отрицателни и големи в абсолютно изражение стойности на
отклонението от потенциалното производство, икономиката се намира в \emph{рецесия} 
\item За $r_t<\bar{r}$ имаме $\hat{y}_t > 0$, т.е. производството в икономиката е над потенциалното и \emph{икономиката прегрява}
\item Приели сме, че потенциалното производство се влияе от дългосрочни
фактори и не се променя при повишение на реалния лихвен процент 
\item Уравнение \eqref{eq:ISline0} свързва отклонението от потенциалното производство и реалния лихвен процент по линия на търсенето в икономиката (агрегатно търсене, съвкупно търсене)
\item Съответно шоковете $\varepsilon_{1,t}$ са от страна на търсенето
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Основен IS-MP-PC модел}
\framesubtitle{IS връзка}
Алтернативен запис на IS правата във вид на отклонения от равновесните стойности на реалния лихвен процент и отклонението от потенциалния БВП: \beq \hat{y}_t = -a( r_t-\bar{r})+\varepsilon_{1,t}. \label{eq:ISline}\eeq
\end{frame}

\begin{frame}
\frametitle{Основен IS-MP-PC модел}
\framesubtitle{PC връзка}
\begin{itemize}
\item Предлагането в икономиката моделираме с \emph{{крива на Филипс}}
\item Тя свързва темпа на инфлация с отклонението от потенциалното производство
\item В случая връзката е линейна (имаме права)
\end{itemize}

\medskip\textbf{PC права}:
\beq
\pi_t=\bar{\pi}+b\hat{y}_t+\varepsilon_{2,t},\label{eq:PCline}\eeq
където $\bar{\pi}$ -- равновесен темп на инфлация, $\varepsilon_{2,t}$ -- шок от страна на предлагането, а $b>0$ е коефициент
\end{frame}

\begin{frame}
\frametitle{Основен IS-MP-PC модел}
\framesubtitle{PC връзка}
\begin{itemize}
\item Когато производството е равно на потенциалното и отсъстват шокове,
имаме $\pi_t=\bar{\pi}$ 
\item Поради това $\bar{\pi}$ се интерпретира като равновесна инфлация
\item Уравнение \eqref{eq:PCline} представлява положителна
връзка на ценови показател с показател за реално производство $\Rightarrow$
аналогия със стандартната крива на предлагане от микроикономиката
\item Шокове в предлагането -- петролна криза, лоша реколта, ...
\item Има и по-сложни формулировки на правата PC, напр. когато миналите стойности на инфлацията влияят върху текущата \'и стойност: \beq \pi_t-\bar{\pi}=c (\pi_{t-1}-\bar{\pi})+b\hat{y}_t+\varepsilon_{2,t},\label{eq:PClineDyn}\eeq
където $c$ е подходящо избран коефициент
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Класическа крива на Филипс}
\framesubtitle{}
\begin{itemize}
\item Терминът \emph{крива на Филипс} често се използва за случая на отрицателна връзка между инфлацията и коефициента на безработица
\item Това може да се получи като следствие от използваната тук постановка
\item Използваме линейна производствена функция с един фактор на производство -- труд: $$Y_t = A L_t$$
\item На потенциалното производство $\bar{Y}_t$ съответства ниво на потенциална заетост $\bar{L}_t$
\item Тогава имаме $\bar{Y}_t = A \bar{L}_t$ и оттам $\hat{y}_t = \hat{L}_t$, където $\hat{L}_t := L_t/\bar{L}_t-1$
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Класическа крива на Филипс}
\framesubtitle{}
\begin{itemize}
\item $LF_t$ -- работната сила в период $t$
\item Приемаме, че тя е винаги на равновесното си равнище: $LF_t \equiv \overline{{LF}}_t$
\item Имаме $$LF_t = L_t + U_t,$$ където $U_t$ е броят на безработните в
разглежданата икономика
\item $u_t := U_t/LF_t$ -- коефициент на безработица
\item Ако икономиката е в равновесие: $$LF_t = \bar{L}_t + \bar{U}_t$$ 
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Класическа крива на Филипс}
\framesubtitle{}
\begin{itemize}
\item В равновесие дяловете на заетостта и безработицата в работната сила са постоянни:
$$\bar{u}_t=\dfrac{\bar{U}_t}{\overline{LF}_t}=\bar{u}$$ 
\item Равновесният коефициент на безработица $\bar{u}$ се нарича \emph{естествена норма на безработица}
\item Разликата $U_t-\bar{U}_t$ се нарича \emph{циклична безработица}
\item Дори в равновесие естествената норма на безработица е положителна
\begin{itemize}
\item Фрикционна безработица
\item Негъвкавост в заплатите
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Класическа крива на Филипс}
\framesubtitle{}
\begin{itemize}
\item Имаме $$LF_t=\bar{L}_t(1+\hat{L}_t)+U_t \quad \Rightarrow \quad
1=(1-\bar{u})(1+\hat{L}_t)+ u_t$$ 
\item Това може да бъде представено като
$$\hat{L}_t = \frac{\bar{u}-u_t}{1-\bar{u}} (=\hat{y}_t)$$ 
\item Замествайки последния израз в PC правата \eqref{eq:PCline}, получаваме отрицателна връзка между инфлация и безработица
$$\pi_t=\bar{\pi}+b \frac{\bar{u}-u_t}{1-\bar{u}} +\varepsilon_{2,t}$$ 
\item Това е алтернативна форма на ,,кривата'' на Филипс -- \emph{класическа крива на Филипс}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Основен IS-MP-PC модел}
\framesubtitle{MP връзка}
\begin{itemize}
\item Последната връзка в IS-MP-PC модела отразява паричната политика
\item Приемаме, че централната банка може да контролира достатъчно добре реалния лихвен процент
\item При провеждането на парична политика лихвеният процент се променя при изменения в избрани макроикономически величини
\item В случая: при промени в отклонението от потенциалното производство и инфлацията
\end{itemize}
$$r_t=f(\hat{y}_t,\pi_t).$$
\end{frame}

\begin{frame}
\frametitle{Основен IS-MP-PC модел}
\framesubtitle{MP връзка}
\begin{itemize}
\item Желателно е паричната политика да не вади икономиката от състояние на равновесие
\item Ако производството се намира на потенциала и
инфлацията е на равновесната си стойност, трябва реалният лихвен процент да е равен на равновесния
$$r_t=\bar{r}=f(0,\bar{\pi})$$
\item В линейния случай имаме \textbf{MP права}
\[ r_t=\bar{r}+d(\pi_t-\bar{\pi})+e \hat{y}_t,  \] където $d,e>0$
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Политика на постоянен реален лихвен процент}
\framesubtitle{}
\begin{itemize}
\item Разглеждаме най-простия случай, когато централната банка поддържа реалния лихвен процент постоянен на равнище $r_t\equiv \bar{r}$
\item Този случай не е реалистичен от практическа гледна точка... \pause
\item ...но дава добра база за сравнение с други, активни политики
\item Ще разгледаме ефектите от шокове в търсенето и в предлагането
\item Стандартно изследваме положителни шокове (в линейния случай, с точност до знака, анализът е аналогичен за отрицателни шокове) \pause
\item {\color{red} ,,положителен'' шок $\neq$ ,,желан'' шок}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Политика на постоянен реален лихвен процент}
\framesubtitle{Шок в търсенето}
\begin{itemize}
\item В период $t$ имаме шок в търсенето: $\varepsilon_{1,t}>0$
(като при това $\varepsilon_{2,t}=0$)
\item От IS връзката получаваме
$$\hat{y}_t = \varepsilon_{1,t}\quad (>0)$$ 
\item Заместваме в PC връзката: $$\pi_t = \bar{\pi} + b
\varepsilon_{1,t}\quad (>\bar{\pi})$$ 
\item Интерпретация: при отсъствие на реакция в
реалния лихвен процент, положителен шок в търсенето води до прегряване на икономиката и ускоряване на инфлацията над равновесния \'и темп.
\end{itemize}
\end{frame}

равновесното ниво), а шокът се проявява в нарастване на инфлацията

\end{itemize}

\begin{frame}
\frametitle{Оптимална парична политика}
\framesubtitle{}
\begin{itemize}
\item Централните банки обикновено са активни в политиката си
\item Действията им са ориентирани към постигане на определени цели
\item Най-често това е поддържане на макроикономическа стабилност
\item В случая макроикономическа стабилност означава поддържане на инфлацията и производството близо до равновесните им стойности
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Оптимална парична политика}
\framesubtitle{}
\begin{itemize}
\item Формализация на целта: минимизира се избрана \emph{функция на загубите}
\item Разглеждаме квадратична функция на загубите, при която централната
банка има собствена цел за инфлацията, $\pi^T$, и цел за потенциалния продукт $\bar{Y}_t$, т.е. $\hat{y}_t=0$
\item Целевата функция е от типа \emph{функция на загубите} \beq L=(\pi_t-\pi^T)^2+\lambda \hat{y}_t^2 ,
\label{eq:QuadrLoss}\eeq където $\lambda > 0$ отчита относителните
предпочитания на банката между отклоненията от инфлационната цел и
от потенциалното производство
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Оптимална парична политика}
\framesubtitle{}
\begin{itemize}
\item Централната банка се стреми да минимизира функцията на
загубите \eqref{eq:QuadrLoss} 
\item При това се отчитат ограниченията, зададени от структурата на икономиката
\item IS правата може да се разглежда като връзка, позволяваща на централната банка да определи реалния лихвен процент на базата на подходящо избрано отклонение от потенциалното производство 
\item Тогава на разположение остава PC правата
\item Окончателно, задачата е да се минимизира функцията на загубите \eqref{eq:QuadrLoss} при ограничение PC правата \eqref{eq:PCline}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Оптимална парична политика}
\framesubtitle{}
\begin{itemize}
\item Заместваме PC връзката във функцията на загубите: $$L=(\bar{\pi}+b
\hat{y}_t+\varepsilon_{2,t}-\pi^T)^2+\lambda \hat{y}^2_t$$
\item Диференцираме по $\hat{y}$ и приравняваме на нула: \beq \frac{\partial
L}{\partial \hat{y}_t} = 2b(\bar{\pi}+b
\hat{y}_t+\varepsilon_{2,t}-\pi^T)+2\lambda \hat{y}_t = 0.
\label{eq:FOCquadrLossPC}\eeq
\item След преобразуване получаваме \beq
\pi_t=\pi^T-\frac{\lambda}{b}\hat{y}_t .
\label{eq:OptGapInfl}\eeq 
\item Това е връзка, задаваща оптималното от гледна точка на централната банка съотношение между инфлация и отклонение от потенциалното производство  \item Моделът с оптимална парична политика при функция на загубите \eqref{eq:QuadrLoss} се задава от IS правата, PC правата и \eqref{eq:OptGapInfl}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Оптимална парична политика}
\framesubtitle{Изразяване на променливите в модела като функция на шоковете и
коефициентите}
\begin{itemize}
\item Комбинираме PC връзката и оптималното съотношение $\pi_t$-$\hat{y}_t$:
$$\bar{\pi}+b \hat{y}_t +\varepsilon_{2,t} =
\pi^T-\frac{\lambda}{b}\hat{y}_t,  $$ откъдето \beq \hat{y}_t =
\frac{b}{b^2+\lambda}
(\pi^T-\bar{\pi})-\frac{b}{b^2+\lambda}\varepsilon_{2,t}
\label{eq:OptMPyhatFshocks}\eeq
\item Ако в PC връзката заместим израза за $\hat{y}_t$ от
\eqref{eq:OptMPyhatFshocks}, получаваме \beq \pi_t = \bar{\pi} +
\frac{b^2}{b^2+\lambda}(\pi^T-\bar{\pi})+\left(
1-\frac{b^2}{b^2+\lambda} \right)\varepsilon_{2,t}
\label{eq:OptMPpiFshocks}\eeq
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Оптимална парична политика}
\framesubtitle{Изразяване на променливите в модела като функция на шоковете и
коефициентите}
\begin{itemize}
\item От IS връзката, като използваме
\eqref{eq:OptMPyhatFshocks}, имаме $$a(r_t-\bar{r})= b
\frac{\bar{\pi}-\pi^T+\varepsilon_{2,t}}{b^2+\lambda}+\varepsilon_{1,t}$$
\item Оттук получаваме \beq r_t =
\bar{r}+\frac{b}{a(b^2+\lambda)}(\bar{\pi}-\pi^T)+\frac{1}{a}\varepsilon_{1,t}+\frac{b}{a(b^2+\lambda)}\varepsilon_{2,t}
\label{eq:OptMPrFshocks} \eeq 
\item  Последното показва как реалният лихвен процент зависи от коефициентите и шоковете
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Оптимална парична политика}
\framesubtitle{Допълнителни бележки}
\begin{itemize}
\item В този вариант на модела централната банка успява напълно да неутрализира ефектите от шоковете в търсенето (в \eqref{eq:OptMPyhatFshocks} и \eqref{eq:OptMPpiFshocks} не участва $\varepsilon_{1,t}$) 
\item Шоковете в предлагането се отразяват и на ценовите процеси, и на
производството в икономиката \pause
\item Сегашната постановка допуска ситуации, при които
инфлационната цел на централната банка $\pi^T$ се различава от
равновесната инфлация $\bar{\pi}$
\item Тогава дори без шокове инфлацията ще бъде различна от равновесната и
отклонението от потенциалното производство ще бъде ненулево
(вж. уравнения \eqref{eq:OptMPyhatFshocks} и \eqref{eq:OptMPpiFshocks})
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Оптимална парична политика}
\framesubtitle{Допълнителни бележки}
\begin{itemize}
\item Съответно лихвеният процент ще се различава от равновесния лихвен процент (вж. \eqref{eq:OptMPrFshocks})
\item Колко големи са тези отклонения зависи от предпочитанията на централната банка, изразени чрез коефициента $\lambda$
\item Ако $\lambda$ е близко до нула $\longrightarrow$ голяма
тежест на инфлационната  цел в сравнение с целта да поддържа
производството в икономиката близко до потенциалното 
\item При отсъствие на шокове $\pi_t$ ще е близко до $\pi^T$, а
лихвеният процент $r_t$ и отклонението от потенциалното производство
$\hat{y}_t$ ще се отличават от равновесните си стойности съответно с
$\frac{b}{a(b^2+\lambda)}(\bar{\pi}-\pi^T)$ и $\frac{b}{b^2+\lambda}
(\pi^T-\bar{\pi})$
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Оптимална парична политика}
\framesubtitle{Допълнителни бележки}
\begin{itemize}
\item Ако стойностите на $\lambda$ са големи, централната банка
дава приоритет на поддържането на производството близко до
потенциалното
\item Тогава стойностите на инфлацията и реалния лихвен
процент при отсъствие на шокове ще са близки съответно до
$\bar{\pi}$ и $\bar{r}$
\end{itemize}\pause

\begin{frame}
\frametitle{Оптимална парична политика}
\framesubtitle{Допълнителни бележки}

{\color{red} Не, избраната функция на загубите третира асиметрично инфлацията и отклонението от потенциалното производство.

Ако във функцията на загубите имаше например
$\lambda \left(\frac{Y_t-Y_t^T}{Y_t^T}\right)^2$ вместо $\lambda
\hat{y}_t$, за някакви целеви стойности на производството $Y_t^T$,
то резултатите няма да изглеждат еднозначно в полза на отклонението
от потенциалното производство.} 
\end{frame}

\begin{frame}
\frametitle{Оптимална парична политика}
\framesubtitle{Допълнителни бележки}
\begin{itemize}
\item Дефинирането на инфлационна цел $\pi^T$, за която евентуално имаме $\pi^T \neq \bar{\pi}$, може да бъде резултат от:
\begin{itemize}
\item отчитане на допълнителни фактори при дефинирането на инфлационната цел (например политически фактори)
\item недостатъчно знание за структурата и характеристиките на икономиката -- централната банка
погрешно смята, че равновесният темп на инфлация е $\pi^T$
\end{itemize}
\item С течение на времето икономическите агенти се адаптират към средата, очакванията им влияят върху равновесния темп на инфлация $\bar{\pi}$ и той се доближава постепенно до $\pi^T$ 
\item Централната банка също би могла да коригира инфлационната си цел
към $\bar{\pi}$, ако познанията \'и за структурата на икономиката се
подобряват с течение на времето
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Оптимална парична политика}
\framesubtitle{Извеждане на MP права}
\begin{itemize}
\item Комбинираме IS връзката и PC връзката:
$$ \pi_t=\bar{\pi}-ab(r_t-\bar{r})+b
\varepsilon_{1,t}+\varepsilon_{2,t}$$ 
\item Заместваме този израз във връзката за оптималното съотношение $\pi_t$-$\hat{y}_t$:
$$\pi^T-\frac{\lambda}{b}\hat{y}_t=\bar{\pi}-ab(r_t-\bar{r})+b \varepsilon_{1,t}+\varepsilon_{2,t}$$
\item След преобразувания получаваме \emph{MP правата}: \beq
r_t-\bar{r}=\frac{\bar{\pi}-\pi^T}{a b}+\frac{\lambda}{ab^2}\hat{y}_t
+\frac{1}{a}\varepsilon_{1,t}+\frac{1}{a b}\varepsilon_{2,t}
\label{eq:OptMPline}\eeq 
\item Тази права има положителен наклон
\end{itemize}
\end{frame}
